CREATE TABLE O_RollCall(
    RollNo INT PRIMARY KEY,
    Name VARCHAR(50)
);

--------------------------------------------------
-- Create New Roll Call Table
--------------------------------------------------

CREATE TABLE N_RollCall(
    RollNo INT PRIMARY KEY,
    Name VARCHAR(50)
);

--------------------------------------------------
-- Create Procedure using Cursor
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE MergeRollCall()

BEGIN

    --------------------------------------------------
    -- Variable Declaration
    --------------------------------------------------

    DECLARE done INT DEFAULT 0;

    DECLARE v_rollno INT;
    DECLARE v_name VARCHAR(50);

    DECLARE v_count INT;

    --------------------------------------------------
    -- Cursor Declaration
    --------------------------------------------------

    DECLARE c1 CURSOR FOR
    SELECT RollNo, Name
    FROM N_RollCall;

    --------------------------------------------------
    -- Handler Declaration
    --------------------------------------------------

    DECLARE CONTINUE HANDLER
    FOR NOT FOUND
    SET done = 1;

    --------------------------------------------------
    -- Open Cursor
    --------------------------------------------------

    OPEN c1;

    read_loop: LOOP

        --------------------------------------------------
        -- Fetch Data
        --------------------------------------------------

        FETCH c1 INTO v_rollno, v_name;

        --------------------------------------------------
        -- Exit Condition
        --------------------------------------------------

        IF done = 1 THEN
            LEAVE read_loop;
        END IF;

        --------------------------------------------------
        -- Check Record Exists
        --------------------------------------------------

        SELECT COUNT(*)
        INTO v_count
        FROM O_RollCall
        WHERE RollNo = v_rollno;

        --------------------------------------------------
        -- Insert if Not Exists
        --------------------------------------------------

        IF v_count = 0 THEN

            INSERT INTO O_RollCall
            VALUES(v_rollno, v_name);

            SELECT CONCAT
            (
                'Record Inserted: ',
                v_rollno,
                ' ',
                v_name
            ) AS Message;

        ELSE

            SELECT CONCAT
            (
                'Record Already Exists: ',
                v_rollno
            ) AS Message;

        END IF;

    END LOOP;

    --------------------------------------------------
    -- Close Cursor
    --------------------------------------------------

    CLOSE c1;

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL MergeRollCall();

--------------------------------------------------
-- Display Final Table
--------------------------------------------------

SELECT * FROM O_RollCall;
