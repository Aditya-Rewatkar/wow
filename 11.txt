--------------------------------------------------
-- Create Table
--------------------------------------------------

CREATE TABLE areas(
    radius INT,
    area DECIMAL(10,2)
);

--------------------------------------------------
-- Create Stored Procedure
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE calc_area()

BEGIN

    DECLARE r INT;
    DECLARE a DECIMAL(10,2);

    SET r = 5;

    WHILE r <= 9 DO

        SET a = 3.14 * r * r;

        INSERT INTO areas(radius, area)
        VALUES(r, a);

        SET r = r + 1;

    END WHILE;

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL calc_area();

--------------------------------------------------
-- Display Table Data
--------------------------------------------------

SELECT * FROM areas;
