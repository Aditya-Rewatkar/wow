--------------------------------------------------
-- Create Function to Return Years
--------------------------------------------------

DELIMITER //

CREATE FUNCTION Age_calc(dob DATE)

RETURNS INT

DETERMINISTIC

BEGIN

    DECLARE years INT;

    --------------------------------------------------
    -- Calculate Years
    --------------------------------------------------

    SET years =
    TIMESTAMPDIFF(YEAR, dob, CURDATE());

    RETURN years;

END //

DELIMITER ;

--------------------------------------------------
-- Create Procedure for
-- Years, Months and Days
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE Age_Details
(
    IN dob DATE,
    OUT years INT,
    OUT months INT,
    OUT days INT
)

BEGIN

    --------------------------------------------------
    -- Calculate Years
    --------------------------------------------------

    SET years =
    TIMESTAMPDIFF(YEAR, dob, CURDATE());

    --------------------------------------------------
    -- Calculate Months
    --------------------------------------------------

    SET months =
    TIMESTAMPDIFF
    (
        MONTH,
        DATE_ADD(dob,
        INTERVAL years YEAR),
        CURDATE()
    );

    --------------------------------------------------
    -- Calculate Days
    --------------------------------------------------

    SET days =
    DATEDIFF
    (
        CURDATE(),
        DATE_ADD
        (
            DATE_ADD(dob,
            INTERVAL years YEAR),
            INTERVAL months MONTH
        )
    );

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL Age_Details
(
    '2003-05-15',
    @y,
    @m,
    @d
);

--------------------------------------------------
-- Display Output
--------------------------------------------------

SELECT
@y AS Years,
@m AS Months,
@d AS Days;

--------------------------------------------------
-- Execute Function
--------------------------------------------------

SELECT Age_calc('2003-05-15')
AS Total_Years;
