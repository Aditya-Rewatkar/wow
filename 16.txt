--------------------------------------------------
-- Create Stored Function Age_calc
--------------------------------------------------

CREATE OR REPLACE FUNCTION Age_calc
(
    dob IN DATE,
    months OUT NUMBER,
    days OUT NUMBER
)

RETURN NUMBER

IS

    years NUMBER;

BEGIN

    --------------------------------------------------
    -- Calculate Years
    --------------------------------------------------

    years := FLOOR(MONTHS_BETWEEN(SYSDATE, dob) / 12);

    --------------------------------------------------
    -- Calculate Remaining Months
    --------------------------------------------------

    months := FLOOR
    (
        MOD(MONTHS_BETWEEN(SYSDATE, dob), 12)
    );

    --------------------------------------------------
    -- Calculate Remaining Days
    --------------------------------------------------

    days := FLOOR
    (
        SYSDATE -
        ADD_MONTHS
        (
            dob,
            years * 12 + months
        )
    );

    --------------------------------------------------
    -- Return Years
    --------------------------------------------------

    RETURN years;

END;
/



DECLARE

    y NUMBER;
    m NUMBER;
    d NUMBER;

BEGIN

    y := Age_calc
    (
        TO_DATE('15-05-2003','DD-MM-YYYY'),
        m,
        d
    );

    DBMS_OUTPUT.PUT_LINE
    ('Age = ' || y || ' Years '
                 || m || ' Months '
                 || d || ' Days');

END;
/
