--------------------------------------------------
-- Create Borrower Table
--------------------------------------------------

CREATE TABLE Borrower(
    Rollin INT PRIMARY KEY,
    Name VARCHAR(50),
    DateofIssue DATE,
    NameofBook VARCHAR(50),
    Status CHAR(1)
);

--------------------------------------------------
-- Create Fine Table
--------------------------------------------------

CREATE TABLE Fine(
    Roll_no INT,
    Date1 DATE,
    Amt DECIMAL(10,2)
);

--------------------------------------------------
-- Create Stored Procedure
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE LibraryFine
(
    IN r_no INT,
    IN b_name VARCHAR(50)
)

BEGIN

    --------------------------------------------------
    -- Variable Declaration
    --------------------------------------------------

    DECLARE issue_date DATE;
    DECLARE no_of_days INT;
    DECLARE fine_amt DECIMAL(10,2) DEFAULT 0;

    --------------------------------------------------
    -- Get Issue Date
    --------------------------------------------------

    SELECT DateofIssue
    INTO issue_date
    FROM Borrower
    WHERE Rollin = r_no
    AND NameofBook = b_name;

    --------------------------------------------------
    -- Calculate Number of Days
    --------------------------------------------------

    SET no_of_days =
    DATEDIFF(CURDATE(), issue_date);

    --------------------------------------------------
    -- Fine Calculation
    --------------------------------------------------

    IF no_of_days BETWEEN 15 AND 30 THEN

        SET fine_amt = no_of_days * 5;

    ELSEIF no_of_days > 30 THEN

        SET fine_amt =
        (30 * 5) +
        ((no_of_days - 30) * 50);

    END IF;

    --------------------------------------------------
    -- Insert Fine Details
    --------------------------------------------------

    IF fine_amt > 0 THEN

        INSERT INTO Fine
        VALUES(r_no, CURDATE(), fine_amt);

    END IF;

    --------------------------------------------------
    -- Update Book Status
    --------------------------------------------------

    UPDATE Borrower
    SET Status = 'R'
    WHERE Rollin = r_no
    AND NameofBook = b_name;

    --------------------------------------------------
    -- Display Output
    --------------------------------------------------

    SELECT
    no_of_days AS "Number of Days",
    fine_amt AS "Fine Amount",
    'Book Returned Successfully'
    AS Message;

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL LibraryFine(1,'DBMS');

--------------------------------------------------
-- Display Fine Table
--------------------------------------------------

SELECT * FROM Fine;
