--------------------------------------------------
-- Create CUSTOMERS Table
--------------------------------------------------

CREATE TABLE CUSTOMERS(
    Cust_id INT PRIMARY KEY,
    Cust_name VARCHAR(50),
    Salary DECIMAL(10,2)
);

--------------------------------------------------
-- Insert Sample Records
--------------------------------------------------

INSERT INTO CUSTOMERS VALUES
(101,'Aditya',30000);

INSERT INTO CUSTOMERS VALUES
(102,'Rahul',40000);

--------------------------------------------------
-- BEFORE INSERT Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_customer_insert

BEFORE INSERT
ON CUSTOMERS

FOR EACH ROW

BEGIN

    SELECT CONCAT
    (
        'New Salary = ',
        NEW.Salary
    ) AS Message;

END //

DELIMITER ;

--------------------------------------------------
-- BEFORE UPDATE Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_customer_update

BEFORE UPDATE
ON CUSTOMERS

FOR EACH ROW

BEGIN

    SELECT CONCAT
    (
        'Old Salary = ',
        OLD.Salary
    ) AS Old_Salary;

    SELECT CONCAT
    (
        'New Salary = ',
        NEW.Salary
    ) AS New_Salary;

    SELECT CONCAT
    (
        'Salary Difference = ',
        (NEW.Salary - OLD.Salary)
    ) AS Difference;

END //

DELIMITER ;

--------------------------------------------------
-- BEFORE DELETE Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_customer_delete

BEFORE DELETE
ON CUSTOMERS

FOR EACH ROW

BEGIN

    SELECT CONCAT
    (
        'Deleted Salary = ',
        OLD.Salary
    ) AS Message;

END //

DELIMITER ;
