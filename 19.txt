--------------------------------------------------
-- Create Employee Table
--------------------------------------------------

CREATE TABLE Emp(
    Emp_no NUMBER PRIMARY KEY,
    Emp_name VARCHAR2(50),
    Emp_salary NUMBER
);

--------------------------------------------------
-- Create Tracking Table
--------------------------------------------------

CREATE TABLE Tracking(
    Emp_no NUMBER,
    Emp_salary NUMBER
);

--------------------------------------------------
-- Insert Sample Records
--------------------------------------------------

INSERT INTO Emp VALUES
(101,'Aditya',60000);

INSERT INTO Emp VALUES
(102,'Rahul',70000);

COMMIT;

--------------------------------------------------
-- Create AFTER Trigger
--------------------------------------------------

CREATE OR REPLACE TRIGGER trg_emp_salary
AFTER INSERT OR UPDATE OR DELETE
ON Emp
FOR EACH ROW

BEGIN

    --------------------------------------------------
    -- INSERT Operation
    --------------------------------------------------

    IF INSERTING THEN

        IF :NEW.Emp_salary < 50000 THEN

            DBMS_OUTPUT.PUT_LINE
            ('Salary less than 50000 inserted');

            INSERT INTO Tracking
            VALUES(:NEW.Emp_no, :NEW.Emp_salary);

        END IF;

    --------------------------------------------------
    -- UPDATE Operation
    --------------------------------------------------

    ELSIF UPDATING THEN

        IF :NEW.Emp_salary < 50000 THEN

            DBMS_OUTPUT.PUT_LINE
            ('Updated salary less than 50000');

            INSERT INTO Tracking
            VALUES(:NEW.Emp_no, :NEW.Emp_salary);

        END IF;

    --------------------------------------------------
    -- DELETE Operation
    --------------------------------------------------

    ELSIF DELETING THEN

        DBMS_OUTPUT.PUT_LINE
        ('Employee Record Deleted');

    END IF;

END;
/
