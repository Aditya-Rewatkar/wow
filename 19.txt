--------------------------------------------------
-- Create Employee Table
--------------------------------------------------

CREATE TABLE Emp(
    Emp_no INT PRIMARY KEY,
    Emp_name VARCHAR(50),
    Emp_salary DECIMAL(10,2)
);

--------------------------------------------------
-- Create Tracking Table
--------------------------------------------------

CREATE TABLE Tracking(
    Emp_no INT,
    Emp_salary DECIMAL(10,2)
);

--------------------------------------------------
-- Insert Sample Records
--------------------------------------------------

INSERT INTO Emp VALUES
(101,'Aditya',60000);

INSERT INTO Emp VALUES
(102,'Rahul',70000);

--------------------------------------------------
-- AFTER INSERT Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_emp_insert

AFTER INSERT
ON Emp

FOR EACH ROW

BEGIN

    IF NEW.Emp_salary < 50000 THEN

        INSERT INTO Tracking
        VALUES(NEW.Emp_no, NEW.Emp_salary);

        SELECT
        'Salary less than 50000 inserted'
        AS Message;

    END IF;

END //

DELIMITER ;

--------------------------------------------------
-- AFTER UPDATE Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_emp_update

AFTER UPDATE
ON Emp

FOR EACH ROW

BEGIN

    IF NEW.Emp_salary < 50000 THEN

        INSERT INTO Tracking
        VALUES(NEW.Emp_no, NEW.Emp_salary);

        SELECT
        'Updated salary less than 50000'
        AS Message;

    END IF;

END //

DELIMITER ;

--------------------------------------------------
-- AFTER DELETE Trigger
--------------------------------------------------

DELIMITER //

CREATE TRIGGER trg_emp_delete

AFTER DELETE
ON Emp

FOR EACH ROW

BEGIN

    SELECT
    'Employee Record Deleted'
    AS Message;

END //

DELIMITER ;
