--------------------------------------------------
-- Create Employee Table
--------------------------------------------------

CREATE TABLE emp(
    emp_no INT PRIMARY KEY,
    salary DECIMAL(10,2)
);

--------------------------------------------------
-- Create Increment Salary Table
--------------------------------------------------

CREATE TABLE increment_salary(
    emp_no INT,
    salary DECIMAL(10,2)
);

--------------------------------------------------
-- Create Procedure
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE IncreaseSalary()

BEGIN

    --------------------------------------------------
    -- Variable Declaration
    --------------------------------------------------

    DECLARE done INT DEFAULT 0;

    DECLARE v_empno INT;
    DECLARE v_salary DECIMAL(10,2);

    DECLARE avg_salary DECIMAL(10,2);

    --------------------------------------------------
    -- Cursor Declaration
    --------------------------------------------------

    DECLARE c1 CURSOR FOR

    SELECT emp_no, salary
    FROM emp
    WHERE salary < avg_salary;

    --------------------------------------------------
    -- Handler Declaration
    --------------------------------------------------

    DECLARE CONTINUE HANDLER
    FOR NOT FOUND
    SET done = 1;

    --------------------------------------------------
    -- Calculate Average Salary
    --------------------------------------------------

    SELECT AVG(salary)
    INTO avg_salary
    FROM emp;

    --------------------------------------------------
    -- Open Cursor
    --------------------------------------------------

    OPEN c1;

    read_loop: LOOP

        --------------------------------------------------
        -- Fetch Record
        --------------------------------------------------

        FETCH c1 INTO v_empno, v_salary;

        --------------------------------------------------
        -- Exit Condition
        --------------------------------------------------

        IF done = 1 THEN
            LEAVE read_loop;
        END IF;

        --------------------------------------------------
        -- Increase Salary by 10%
        --------------------------------------------------

        UPDATE emp
        SET salary = salary + (salary * 0.10)
        WHERE emp_no = v_empno;

        --------------------------------------------------
        -- Store Updated Salary
        --------------------------------------------------

        INSERT INTO increment_salary
        VALUES
        (
            v_empno,
            v_salary + (v_salary * 0.10)
        );

        --------------------------------------------------
        -- Display Message
        --------------------------------------------------

        SELECT CONCAT
        (
            'Salary Updated for Employee No: ',
            v_empno
        ) AS Message;

    END LOOP;

    --------------------------------------------------
    -- Close Cursor
    --------------------------------------------------

    CLOSE c1;

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL IncreaseSalary();

--------------------------------------------------
-- Display Updated Employee Table
--------------------------------------------------

SELECT * FROM emp;

--------------------------------------------------
-- Display Increment Salary Table
--------------------------------------------------

SELECT * FROM increment_salary;
