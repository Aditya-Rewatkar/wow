--------------------------------------------------
-- Create Student Table
--------------------------------------------------

CREATE TABLE Student(
    Roll INT PRIMARY KEY,
    Name VARCHAR(50),
    Attendance INT,
    Status VARCHAR(20)
);

--------------------------------------------------
-- Create Procedure
--------------------------------------------------

DELIMITER //

CREATE PROCEDURE CheckAttendance
(
    IN r_no INT
)

BEGIN

    --------------------------------------------------
    -- Variable Declaration
    --------------------------------------------------

    DECLARE att INT;

    --------------------------------------------------
    -- Exception Handling Variable
    --------------------------------------------------

    DECLARE student_count INT;

    --------------------------------------------------
    -- Check Student Exists
    --------------------------------------------------

    SELECT COUNT(*)
    INTO student_count
    FROM Student
    WHERE Roll = r_no;

    --------------------------------------------------
    -- If Student Not Found
    --------------------------------------------------

    IF student_count = 0 THEN

        SELECT
        'Student Roll Number Not Found'
        AS Message;

    ELSE

        --------------------------------------------------
        -- Fetch Attendance
        --------------------------------------------------

        SELECT Attendance
        INTO att
        FROM Student
        WHERE Roll = r_no;

        --------------------------------------------------
        -- Attendance Check
        --------------------------------------------------

        IF att < 75 THEN

            UPDATE Student
            SET Status = 'Detained'
            WHERE Roll = r_no;

            SELECT
            'Term not granted'
            AS Message;

        ELSE

            UPDATE Student
            SET Status = 'Not Detained'
            WHERE Roll = r_no;

            SELECT
            'Term granted'
            AS Message;

        END IF;

    END IF;

END //

DELIMITER ;

--------------------------------------------------
-- Execute Procedure
--------------------------------------------------

CALL CheckAttendance(2);
